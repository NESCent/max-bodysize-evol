  def test_presentation_style
    # default
    assert @link.inline?
    assert !@link.popup?
    assert !@link.page?

    @link.popup = true
    assert !@link.inline?
    assert @link.popup?
    assert !@link.page?

    @link.page = true
    assert !@link.inline?
    assert !@link.popup?
    assert @link.page?

    @link.inline = true
    assert @link.inline?
    assert !@link.popup?
    assert !@link.page?
  end
