  def test_position
    @link.position = true

    @link.type = :table
    assert_equal :top, @link.position

    @link.type = :record
    assert_equal :replace, @link.position

    @link.position = :before
    assert_equal :before, @link.position

    @link.position = false
    assert_equal false, @link.position
  end
