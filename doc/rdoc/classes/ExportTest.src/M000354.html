    def test_execute()
        require 'fastercsv'
        @export.execute()
        row_length = @export.header.length()

        # Parse the csv_string.
        # Each row should have exactly as many elements as the header.
        FasterCSV.parse(@export.csv_string, :headers => true) do |row|
            assert_equal(row_length, row.length)
        end
    end
