  def test_sorts_by_method
    @columns[:a].sort_by :method => proc{0}

    #test pure method sorting: true
    @sorting.add :a
    assert @sorting.sorts_by_method?

    #test mixed sql/method sorting: true
    @sorting.add :b
    assert @sorting.sorts_by_method?

    #test pure sql sorting: false
    @sorting.clear
    @sorting.add :b
    assert !@sorting.sorts_by_method?
  end
