  def approve
    @bodysize = Bodysize.find_by_id(params[:id])
    
    if @current_logged_in_user.is_a?(Approver)
      @bodysize.approved = true
      @bodysize.save
      AuditLog.log("Approved a bodysize record", @current_logged_in_user.id, @bodysize.id)
    end
  end
