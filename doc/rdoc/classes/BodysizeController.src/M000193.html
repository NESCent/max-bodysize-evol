  def remove_comment
    return unless authorize_admin
    
    comment = Comment.find_by_id(params[:id])
    
    if comment
      bodysize = comment.bodysize
      comment.destroy
      
      render :partial => "comments", :locals => { :bodysize => bodysize }  
      return
    end
    
    render :text => "The selected comment could not be found"
  end
