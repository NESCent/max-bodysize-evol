    def create_labels(periods, period_spacing)
      labels = Array.new
      label_number = 0
      periods.each do |period|
        if period_spacing == "midpoint"
          label = Label.new(period.name, period.midpoint)
        else
          label = Label.new(period.name, periods.length - label_number - 1)
        end
        
        labels << label
        label_number += 1
      end
      
      return labels
    end
