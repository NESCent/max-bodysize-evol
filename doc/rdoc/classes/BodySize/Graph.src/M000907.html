    def get_data(bodysizes, grouping_field)
      results = Hash.new
      bodysizes.each do |specimen|
        field_value = specimen.send(grouping_field.attribute_name)
        results[field_value] ||= Array.new
        results[field_value] << specimen
      end
      
      return results
    end
