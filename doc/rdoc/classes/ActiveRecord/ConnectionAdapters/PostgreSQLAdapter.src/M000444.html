      def view_select_statement(view, name = nil)
        q = "SELECT view_definition\nFROM information_schema.views\nWHERE table_catalog = (SELECT catalog_name FROM information_schema.information_schema_catalog_name)\nAND table_schema IN (\#{schemas})\nAND table_name = '\#{view}'\n"
        
        select_value(q, name) or raise "No view called #{view} found"
      end
