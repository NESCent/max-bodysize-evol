    def dump_with_views(stream)
      dump_without_views(stream)
      begin
        if @connection.supports_views?
          views(stream)
        end
      rescue => e
        ActiveRecord::Base.logger.error "Unable to dump views: #{e}"
      end
      trailer_without_views(stream)
      stream
    end
