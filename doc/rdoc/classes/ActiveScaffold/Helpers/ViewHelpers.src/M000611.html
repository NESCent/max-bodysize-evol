      def column_empty?(column_value)
        empty = column_value.nil?
        empty ||= column_value.empty? if column_value.respond_to? :empty?
        empty ||= (column_value == '&nbsp;')
        empty ||= (column_value == active_scaffold_config.list.empty_field_text)
        return empty
      end
