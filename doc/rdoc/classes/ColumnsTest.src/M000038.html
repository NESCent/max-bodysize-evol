  def test_add
    assert !@columns.include?(:c)
    @columns.add 'c'
    assert @columns.include?('c')

    # test the alias
    assert !@columns.include?(:d)
    @columns << :d
    assert @columns.include?(:d)

    # try adding an array of columns
    assert !@columns.include?(:f)
    @columns.add [:f, :g]
    assert @columns.include?(:f)
    assert @columns.include?(:g)
  end
