  def self.unapproved_bodysize_records(page = nil)
    if page == nil
      Bodysize.find(:all, :conditions => ["approved=?", false], :order => "bodysizes.updated_at DESC")
    else
      Bodysize.paginate(:all, :conditions => ["approved=?", false], :order => "bodysizes.updated_at DESC", :page => page)
    end
  end
